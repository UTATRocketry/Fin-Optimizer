% Given info
N = 1000;
h = 0.01;
v = 1;
w_min = -pi/4;
w_max = pi/4;
y0 = 1.5;
h0 = 0;
r = 250;
Q  = [250 0; 0 250];

% Cost function - quadprog has to take in f in the form (1/2)*x'*H*x + f'*x

f = zeros(3*N,1);
H = diag(ones(N,1)*r);

for i=1:N
    H = blkdiag(H,Q);
end

% Need to multiply H by 2 since we need to have it in the form that
% quadprog can use (1/2 factor)
H = 2*H;

% Equality Constraints - quadprog uses Ax = b form

% A should be (2N x 3N)
A = zeros(2*N,N); %2N rows since we have N (y,h) pairs
for i=1:N
    A(2*i,i) = -h;
end

% Put together equality constraint matrix for the 2N columns remaining
% based on pattern shown in work
diag1 = [0, repmat([-h*v, 0], 1, N-1)];
diag2 = -1*ones(2*N-2,1);
Temp = eye(2*N) + diag(diag1,-1) + diag(diag2,-2);

A_comp = [A, Temp];
b = [y0+h*v*h0; h0; zeros(2*N-2,1)];

% Inequality Constraints
Temp = [diag(-1*ones(N,1)); diag(ones(N,1))];
B = [Temp zeros(2*N, 2*N)];
c = [-ones(N,1)*w_min; ones(N,1)*w_max];

% Solve
[x, fval,exitflag,output,lambda] = quadprog(H,f,B,c,A_comp,b);
w = x(1:N); % Optimized angular velocity

% Retreive y and h from x
optimal = reshape(x(N+1:3*N), [2,N]);
y_opt = optimal(1,:);
h_opt = optimal(2,:);

% part f) modify position and heading based on noise
y = zeros(1,N+1);
y(1) = y0;

heading = zeros(1,N+1);
heading(1) = h0;

for i=1:N
    y(i+1) = y(i) + h*v*heading(i);
    heading(i+1) = heading(i) + h*w(i) + h*pi/25;
end

%Plot

% Position
figure()
t= (0:N)*h;
fig(1) = subplot(2, 1, 1);
% plot(t,[y0, y_opt])
plot(t,[[y0,y_opt]; y]); % Comparing with noise solution
title('Robot Position')
ylabel('y (m)')
xlabel('time (s)')
grid on

%Heading
fig(2) = subplot(2, 1, 2);
% plot(t,[h0, h_opt])
plot(t,[[h0, h_opt]; heading]);
title('Heading');
ylabel('h (rad)');
xlabel('time (s)')
grid on

linkaxes(fig, 'x');
